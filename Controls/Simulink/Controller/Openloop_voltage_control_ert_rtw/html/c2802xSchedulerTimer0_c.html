130void configureTimer0(float baseRate, float systemClock)



131{



132    /* Initialize CPU Timers here only if Profiling is disabled.



133     * Because if profiling is enabled, CPU Timers are initialized in 



134     * MW_c28xx_board.c in the generated code.



135     */



136	#ifndef MW_EXEC_PROFILER_ON



137	InitCpuTimers();



138	#endif



139



140	/* Configure CPU-Timer 0 to interrupt every 0.0025 sec. */



141	/* Parameters:  Timer Pointer, CPU Freq in MHz, Period in usec. */



142	ConfigCpuTimer(&CpuTimer0, systemClock, baseRate * 1000000);



143	StartCpuTimer0();



144



145	EALLOW;



146	PieVectTable.TINT0 = &TINT0_isr;     /* Hook interrupt to the ISR*/



147	EDIS;



148



149	PieCtrlRegs.PIEIER1.bit.INTx7 = 1;   /* Enable interrupt TINT0 */



150	IER |= M_INT1;



151}

153void enableTimer0Interrupt(void)



154{



155	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



156	PieCtrlRegs.PIEIER1.all |= (1 << 6); /* Enable PIER bit for timer 0 */



157	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



158}



159



160bool disableTimer0Interrupt(void)



161{



162    bool isrEnabled = false;



163	asm(" SETC INTM"); /* Disable Interrupt Global Enable Bit - set it to 1 */



164    isrEnabled = PieCtrlRegs.PIEIER1.all & (1 << 6) ;



165	PieCtrlRegs.PIEIER1.all &= ~(1 << 6); /* Disable PIER bit for timer 0 */



166	asm(" RPT #5 || NOP");



167	IFR &= 0xFFFE;



168	PieCtrlRegs.PIEACK.all = 0x1;



169	asm(" CLRC INTM"); /* Enable Interrupt Global Enable Bit - set it to 0 */



170    return isrEnabled;



171}



172



173void restoreTimer0Interrupt(bool isrStatus){



174    if(isrStatus)



175        enableTimer0Interrupt();



176    else



177        disableTimer0Interrupt();



178}
