17#define ADC_usDELAY  1000L

19//---------------------------------------------------------------------------



20// InitAdc:



21//---------------------------------------------------------------------------



22// This function initializes ADC to a known state.



23//



24// NOTE: ADC INIT IS DIFFERENT ON 2802x DEVICES COMPARED TO OTHER 28X DEVICES



25//



26void InitAdc(void)



27{



28    extern void DSP28x_usDelay(Uint32 Count);



29



30    // *IMPORTANT*



31    // The Device_cal function, which copies the ADC calibration values from TI reserved



32    // OTP into the ADCREFSEL and ADCOFFTRIM registers, occurs automatically in the



33    // Boot ROM. If the boot ROM code is bypassed during the debug process, the



34    // following function MUST be called for the ADC to function according



35    // to specification. The clocks to the ADC MUST be enabled before calling this



36    // function.



37    // See the device data manual and/or the ADC Reference



38    // Manual for more information.



39



40        EALLOW;



41        SysCtrlRegs.PCLKCR0.bit.ADCENCLK = 1;



42        (*Device_cal)();



43        EDIS;



44



45    // To powerup the ADC the ADCENCLK bit should be set first to enable



46    // clocks, followed by powering up the bandgap, reference circuitry, and ADC core.



47    // Before the first conversion is performed a 5ms delay must be observed



48    // after power up to give all analog circuits time to power up and settle



49



50    // Please note that for the delay function below to operate correctly the



51    // CPU_RATE define statement in the F2802x_Examples.h file must



52    // contain the correct CPU clock period in nanoseconds.



53    EALLOW;



54    AdcRegs.ADCCTL1.bit.ADCBGPWD  = 1;      // Power ADC BG



55    AdcRegs.ADCCTL1.bit.ADCREFPWD = 1;      // Power reference



56    AdcRegs.ADCCTL1.bit.ADCPWDN   = 1;      // Power ADC



57    AdcRegs.ADCCTL1.bit.ADCENABLE = 1;      // Enable ADC



58    AdcRegs.ADCCTL1.bit.ADCREFSEL = 0;      // Select internal BG



59    EDIS;



60



61    DELAY_US(ADC_usDELAY);         // Delay before converting ADC channels



62}

64void InitAdcAio()



65{



66



67   EALLOW;



68



69/* Configure ADC pins using AIO regs*/



70// This specifies which of the possible AIO pins will be Analog input pins.



71// NOTE: AIO1,3,5,7-9,11,13,15 are analog inputs in all AIOMUX1 configurations.



72// Comment out other unwanted lines.



73



74    GpioCtrlRegs.AIOMUX1.bit.AIO2 = 2;    // Configure AIO2 for A2 (analog input) operation



75    GpioCtrlRegs.AIOMUX1.bit.AIO4 = 2;    // Configure AIO4 for A4 (analog input) operation



76    GpioCtrlRegs.AIOMUX1.bit.AIO6 = 2;    // Configure AIO6 for A6 (analog input) operation



77    GpioCtrlRegs.AIOMUX1.bit.AIO10 = 2;   // Configure AIO10 for B2 (analog input) operation



78    GpioCtrlRegs.AIOMUX1.bit.AIO12 = 2;   // Configure AIO12 for B4 (analog input) operation



79    GpioCtrlRegs.AIOMUX1.bit.AIO14 = 2;   // Configure AIO14 for B6 (analog input) operation



80



81    EDIS;



82}

84/* AdcoffsetSelfCal-



85   This function re-calibrates the ADC zero offset error by converting the VREFLO reference with



86   the ADC and modifying the ADCOFFTRIM register. VREFLO is sampled by the ADC using an internal



87   MUX select which connects VREFLO to A5 without sacrificing an external ADC pin. This



88   function calls two other functions:



89   - AdcChanSelect(channel) � selects the ADC channel to convert



90   - AdcConversion() � initiates several ADC conversions and returns the average



91*/



92void AdcOffsetSelfCal()



93{



94    Uint16 AdcConvMean;



95    EALLOW;



96    AdcRegs.ADCCTL1.bit.ADCREFSEL = 0;                  //Select internal reference mode



97    AdcRegs.ADCCTL1.bit.VREFLOCONV = 1;                 //Select VREFLO internal connection on B5



98    AdcChanSelect(13);                                  //Select channel B5 for all SOC



99    AdcRegs.ADCOFFTRIM.bit.OFFTRIM = 80;                //Apply artificial offset (+80) to account for a negative offset that may reside in the ADC core



100    AdcConvMean = AdcConversion();                      //Capture ADC conversion on VREFLO



101    AdcRegs.ADCOFFTRIM.bit.OFFTRIM = 80 - AdcConvMean;  //Set offtrim register with new value (i.e remove artical offset (+80) and create a two's compliment of the offset error)



102    AdcRegs.ADCCTL1.bit.VREFLOCONV = 0;                 //Select external ADCIN5 input pin on B5



103    EDIS;



104}

106/*  AdcChanSelect-



107    This function selects the ADC channel to convert by setting all SOC channel selects to a single channel.



108



109     * IMPORTANT * This function will overwrite previous SOC channel select settings. Recommend saving

117    AdcRegs.ADCSOC3CTL.bit.CHSEL= ch_no;



118    AdcRegs.ADCSOC4CTL.bit.CHSEL= ch_no;



119    AdcRegs.ADCSOC5CTL.bit.CHSEL= ch_no;



120    AdcRegs.ADCSOC6CTL.bit.CHSEL= ch_no;



121    AdcRegs.ADCSOC7CTL.bit.CHSEL= ch_no;



122    AdcRegs.ADCSOC8CTL.bit.CHSEL= ch_no;



123    AdcRegs.ADCSOC9CTL.bit.CHSEL= ch_no;



124    AdcRegs.ADCSOC10CTL.bit.CHSEL= ch_no;

132/* AdcConversion -



133   This function initiates several ADC conversions and returns the average. It uses ADCINT1 and ADCINT2



134   to "ping-pong" between SOC0-7 and SOC8-15 and is referred to as "ping-pong" sampling.



135



136     * IMPORTANT * This function will overwrite previous ADC settings. Recommend saving previous settings.



137*/



138Uint16 AdcConversion(void)



139{



140    Uint16 index, SampleSize, Mean, ACQPS_Value;



141    Uint32 Sum;



142



143    index       = 0;            //initialize index to 0



144    SampleSize  = 256;          //set sample size to 256 (**NOTE: Sample size must be multiples of 2^x where is an integer >= 4)



145    Sum         = 0;            //set sum to 0



146    Mean        = 999;          //initialize mean to known value



147



148    //Set the ADC sample window to the desired value (Sample window = ACQPS + 1)



149    ACQPS_Value = 6;



150    AdcRegs.ADCSOC0CTL.bit.ACQPS  = ACQPS_Value;



151    AdcRegs.ADCSOC1CTL.bit.ACQPS  = ACQPS_Value;



152    AdcRegs.ADCSOC2CTL.bit.ACQPS  = ACQPS_Value;



153    AdcRegs.ADCSOC3CTL.bit.ACQPS  = ACQPS_Value;



154    AdcRegs.ADCSOC4CTL.bit.ACQPS  = ACQPS_Value;



155    AdcRegs.ADCSOC5CTL.bit.ACQPS  = ACQPS_Value;



156    AdcRegs.ADCSOC6CTL.bit.ACQPS  = ACQPS_Value;



157    AdcRegs.ADCSOC7CTL.bit.ACQPS  = ACQPS_Value;



158    AdcRegs.ADCSOC8CTL.bit.ACQPS  = ACQPS_Value;



159    AdcRegs.ADCSOC9CTL.bit.ACQPS  = ACQPS_Value;



160    AdcRegs.ADCSOC10CTL.bit.ACQPS = ACQPS_Value;



161    AdcRegs.ADCSOC11CTL.bit.ACQPS = ACQPS_Value;



162    AdcRegs.ADCSOC12CTL.bit.ACQPS = ACQPS_Value;



163    AdcRegs.ADCSOC13CTL.bit.ACQPS = ACQPS_Value;



164    AdcRegs.ADCSOC14CTL.bit.ACQPS = ACQPS_Value;



165    AdcRegs.ADCSOC15CTL.bit.ACQPS = ACQPS_Value;



166



167



168    //Enable ping-pong sampling



169



170    // Enabled ADCINT1 and ADCINT2



171    AdcRegs.INTSEL1N2.bit.INT1E = 1;



172    AdcRegs.INTSEL1N2.bit.INT2E = 1;



173



174    // Disable continuous sampling for ADCINT1 and ADCINT2



175    AdcRegs.INTSEL1N2.bit.INT1CONT = 0;



176    AdcRegs.INTSEL1N2.bit.INT2CONT = 0;



177



178    AdcRegs.ADCCTL1.bit.INTPULSEPOS = 1;    //ADCINTs trigger at end of conversion



179



180    // Setup ADCINT1 and ADCINT2 trigger source



181    AdcRegs.INTSEL1N2.bit.INT1SEL = 6;      //EOC6 triggers ADCINT1



182    AdcRegs.INTSEL1N2.bit.INT2SEL = 14;     //EOC14 triggers ADCINT2



183



184    // Setup each SOC's ADCINT trigger source



185    AdcRegs.ADCINTSOCSEL1.bit.SOC0  = 2;    //ADCINT2 starts SOC0-7



186    AdcRegs.ADCINTSOCSEL1.bit.SOC1  = 2;



187    AdcRegs.ADCINTSOCSEL1.bit.SOC2  = 2;



188    AdcRegs.ADCINTSOCSEL1.bit.SOC3  = 2;



189    AdcRegs.ADCINTSOCSEL1.bit.SOC4  = 2;



190    AdcRegs.ADCINTSOCSEL1.bit.SOC5  = 2;



191    AdcRegs.ADCINTSOCSEL1.bit.SOC6  = 2;



192    AdcRegs.ADCINTSOCSEL1.bit.SOC7  = 2;



193    AdcRegs.ADCINTSOCSEL2.bit.SOC8  = 1;    //ADCINT1 starts SOC8-15



194    AdcRegs.ADCINTSOCSEL2.bit.SOC9  = 1;



195    AdcRegs.ADCINTSOCSEL2.bit.SOC10 = 1;



196    AdcRegs.ADCINTSOCSEL2.bit.SOC11 = 1;



197    AdcRegs.ADCINTSOCSEL2.bit.SOC12 = 1;



198    AdcRegs.ADCINTSOCSEL2.bit.SOC13 = 1;



199    AdcRegs.ADCINTSOCSEL2.bit.SOC14 = 1;



200    AdcRegs.ADCINTSOCSEL2.bit.SOC15 = 1;



201



202    DELAY_US(ADC_usDELAY);                  // Delay before converting ADC channels



203



204



205    //ADC Conversion



206



207    AdcRegs.ADCSOCFRC1.all = 0x00FF;  // Force Start SOC0-7 to begin ping-pong sampling



208



209    while( index < SampleSize ){



210



211        //Wait for ADCINT1 to trigger, then add ADCRESULT0-7 registers to sum



212        while (AdcRegs.ADCINTFLG.bit.ADCINT1 == 0){}



213        AdcRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;   //Must clear ADCINT1 flag since INT1CONT = 0



214        Sum += AdcResult.ADCRESULT0;



215        Sum += AdcResult.ADCRESULT1;



216        Sum += AdcResult.ADCRESULT2;



217        Sum += AdcResult.ADCRESULT3;



218        Sum += AdcResult.ADCRESULT4;



219        Sum += AdcResult.ADCRESULT5;



220        Sum += AdcResult.ADCRESULT6;



221        // Wait for SOC9 conversion to start, which gives time for SOC7 conversion result



222        while( AdcRegs.ADCSOCFLG1.bit.SOC9 == 1 ){}



223        Sum += AdcResult.ADCRESULT7;



224



225        //Wait for ADCINT2 to trigger, then add ADCRESULT8-15 registers to sum



226        while (AdcRegs.ADCINTFLG.bit.ADCINT2 == 0){}



227        AdcRegs.ADCINTFLGCLR.bit.ADCINT2 = 1;   //Must clear ADCINT2 flag since INT2CONT = 0



228        Sum += AdcResult.ADCRESULT8;



229        Sum += AdcResult.ADCRESULT9;



230        Sum += AdcResult.ADCRESULT10;



231        Sum += AdcResult.ADCRESULT11;



232        Sum += AdcResult.ADCRESULT12;



233        Sum += AdcResult.ADCRESULT13;



234        Sum += AdcResult.ADCRESULT14;



235        // Wait for SOC1 conversion to start, which gives time for SOC15 conversion result



236        while( AdcRegs.ADCSOCFLG1.bit.SOC1 == 1 ){}



237        Sum += AdcResult.ADCRESULT15;



238



239        index+=16;



240



241    } // end data collection



242



243    //Disable ADCINT1 and ADCINT2 to STOP the ping-pong sampling



244    AdcRegs.INTSEL1N2.bit.INT1E = 0;



245    AdcRegs.INTSEL1N2.bit.INT2E = 0;



246



247    while(AdcRegs.ADCSOCFLG1.all != 0){} // Wait for any pending SOCs to complete



248



249    // Clear any pending interrupts



250    AdcRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;



251    AdcRegs.ADCINTFLGCLR.bit.ADCINT2 = 1;



252    AdcRegs.ADCINTOVFCLR.bit.ADCINT1 = 1;



253    AdcRegs.ADCINTOVFCLR.bit.ADCINT2 = 1;



254



255    //reset RR pointer to 32, so that next SOC is SOC0



256    AdcRegs.SOCPRICTL.bit.SOCPRIORITY = 1;



257    while( AdcRegs.SOCPRICTL.bit.SOCPRIORITY != 1 );



258    AdcRegs.SOCPRICTL.bit.SOCPRIORITY = 0;



259    while( AdcRegs.SOCPRICTL.bit.SOCPRIORITY != 0 );



260



261    Mean = Sum / SampleSize;    //Calculate average ADC sample value



262



263    return Mean;                //return the average



264



265}//end AdcConversion
