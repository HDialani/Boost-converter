27//---------------------------------------------------------------------------



28// InitSysCtrl:



29//---------------------------------------------------------------------------



30// This function initializes the System Control registers to a known state.



31// - Disables the watchdog



32// - Set the PLLCR for proper SYSCLKOUT frequency



33// - Set the pre-scaler for the high and low frequency peripheral clocks



34// - Enable the clocks to the peripherals



35



36void InitSysCtrl(void)



37{



38



39   // Disable the watchdog



40   DisableDog();



41



42    // *IMPORTANT*



43    // The Device_cal function, which copies the ADC & oscillator calibration values



44    // from TI reserved OTP into the appropriate trim registers, occurs automatically



45    // in the Boot ROM. If the boot ROM code is bypassed during the debug process, the



46    // following function MUST be called for the ADC and oscillators to function according



47    // to specification. The clocks to the ADC MUST be enabled before calling this



48    // function.



49    // See the device data manual and/or the ADC Reference



50    // Manual for more information.



51



52        EALLOW;



53        SysCtrlRegs.PCLKCR0.bit.ADCENCLK = 1; // Enable ADC peripheral clock



54        (*Device_cal)();



55        SysCtrlRegs.PCLKCR0.bit.ADCENCLK = 0; // Return ADC clock to original state



56        EDIS;



57



58   // Select Internal Oscillator 1 as Clock Source (default), and turn off all unused clocks to



59   // conserve power.



60   IntOsc1Sel();



61



62   // Initialize the PLL control: PLLCR and DIVSEL



63   // DSP28_PLLCR and DSP28_DIVSEL are defined in F2802x_Examples.h



64   InitPll(DSP28_PLLCR,DSP28_DIVSEL);



65



66   // Initialize the peripheral clocks



67   InitPeripheralClocks();



68}

70//---------------------------------------------------------------------------



71// Example: InitFlash:



72//---------------------------------------------------------------------------



73// This function initializes the Flash Control registers



74



75//                   CAUTION



76// This function MUST be executed out of RAM. Executing it



77// out of OTP/Flash will yield unpredictable results



78



79void InitFlash(void)



80{



81   EALLOW;



82   //Enable Flash Pipeline mode to improve performance



83   //of code executed from Flash.



84   FlashRegs.FOPT.bit.ENPIPE = 1;



85



86   //                CAUTION



87   //Minimum waitstates required for the flash operating



88   //at a given CPU rate must be characterized by TI.



89   //Refer to the datasheet for the latest information.



90



91  #if (CPU_FRQ_60MHZ)



92   //Set the Paged Waitstate for the Flash



93   FlashRegs.FBANKWAIT.bit.PAGEWAIT = 2;



94



95   //Set the Random Waitstate for the Flash



96   FlashRegs.FBANKWAIT.bit.RANDWAIT = 2;



97



98   //Set the Waitstate for the OTP



99   FlashRegs.FOTPWAIT.bit.OTPWAIT = 2;



100



101  #elif (CPU_FRQ_40MHZ)



102   //Set the Paged Waitstate for the Flash



103   FlashRegs.FBANKWAIT.bit.PAGEWAIT = 1;



104



105   //Set the Random Waitstate for the Flash



106   FlashRegs.FBANKWAIT.bit.RANDWAIT = 1;



107



108   //Set the Waitstate for the OTP



109   FlashRegs.FOTPWAIT.bit.OTPWAIT = 1;



110  #endif



111   //                CAUTION



112   //ONLY THE DEFAULT VALUE FOR THESE 2 REGISTERS SHOULD BE USED



113   FlashRegs.FSTDBYWAIT.bit.STDBYWAIT = 0x01FF;



114   FlashRegs.FACTIVEWAIT.bit.ACTIVEWAIT = 0x01FF;



115   EDIS;



116



117   //Force a pipeline flush to ensure that the write to



118   //the last register configured occurs before returning.



119



120   asm(" RPT #7 || NOP");



121}

149//---------------------------------------------------------------------------



150// Example: InitPll:



151//---------------------------------------------------------------------------



152// This function initializes the PLLCR register.



153



154void InitPll(Uint16 val, Uint16 divsel)



155{



156   volatile Uint16 iVol;



157



158   // Make sure the PLL is not running in limp mode



159   if (SysCtrlRegs.PLLSTS.bit.MCLKSTS != 0)



160   {



161      EALLOW;



162      // OSCCLKSRC1 failure detected. PLL running in limp mode.



163      // Re-enable missing clock logic.



164      SysCtrlRegs.PLLSTS.bit.MCLKCLR = 1;



165      EDIS;



166      // Replace this line with a call to an appropriate



167      // SystemShutdown(); function.



168      asm("        ESTOP0");     // Uncomment for debugging purposes



169   }



170



171   // DIVSEL MUST be 0 before PLLCR can be changed from



172   // 0x0000. It is set to 0 by an external reset XRSn



173   // This puts us in 1/4



174   if (SysCtrlRegs.PLLSTS.bit.DIVSEL != 0)



175   {



176       EALLOW;



177       SysCtrlRegs.PLLSTS.bit.DIVSEL = 0;



178       EDIS;



179   }



180



181   // Change the PLLCR



182   if (SysCtrlRegs.PLLCR.bit.DIV != val)



183   {



184



185      EALLOW;



186      // Before setting PLLCR turn off missing clock detect logic



187      SysCtrlRegs.PLLSTS.bit.MCLKOFF = 1;



188      SysCtrlRegs.PLLCR.bit.DIV = val;



189      EDIS;



190



191      // Optional: Wait for PLL to lock.



192      // During this time the CPU will switch to OSCCLK/2 until



193      // the PLL is stable.  Once the PLL is stable the CPU will



194      // switch to the new PLL value.



195      //



196      // This time-to-lock is monitored by a PLL lock counter.



197      //



198      // Code is not required to sit and wait for the PLL to lock.



199      // However, if the code does anything that is timing critical,



200      // and requires the correct clock be locked, then it is best to



201      // wait until this switching has completed.



202



203      // Wait for the PLL lock bit to be set.



204



205      // The watchdog should be disabled before this loop, or fed within



206      // the loop via ServiceDog().



207



208      // Uncomment to disable the watchdog



209      DisableDog();



210



211      while(SysCtrlRegs.PLLSTS.bit.PLLLOCKS != 1)



212      {



213          // Uncomment to service the watchdog



214          // ServiceDog();



215      }



216



217      EALLOW;



218      SysCtrlRegs.PLLSTS.bit.MCLKOFF = 0;



219      EDIS;



220    }



221



222    // If switching to 1/2



223    if((divsel == 1)||(divsel == 2))



224    {



225        EALLOW;



226        SysCtrlRegs.PLLSTS.bit.DIVSEL = divsel;



227        EDIS;



228    }



229



230    // If switching to 1/1



231    // * First go to 1/2 and let the power settle



232    //   The time required will depend on the system, this is only an example



233    // * Then switch to 1/1



234    if(divsel == 3)



235    {



236        EALLOW;



237        SysCtrlRegs.PLLSTS.bit.DIVSEL = 2;



238        DELAY_US(50L);



239        SysCtrlRegs.PLLSTS.bit.DIVSEL = 3;



240        EDIS;



241    }



242}

244//--------------------------------------------------------------------------



245// Example: InitPeripheralClocks:



246//---------------------------------------------------------------------------



247// This function initializes the clocks to the peripheral modules.



248// First the high and low clock prescalers are set



249// Second the clocks are enabled to each peripheral.



250// To reduce power, leave clocks to unused peripherals disabled



251//



252// Note: If a peripherals clock is not enabled then you cannot



253// read or write to the registers for that peripheral



254



255void InitPeripheralClocks(void)



256{



257   EALLOW;



258



259// LOSPCP prescale register settings, normally it will be set to default values



260



261   GpioCtrlRegs.GPAMUX2.bit.GPIO18 = 3;  // GPIO18 = XCLKOUT



262   SysCtrlRegs.LOSPCP.all = 0x0002;



263



264// XCLKOUT to SYSCLKOUT ratio.  By default XCLKOUT = 1/4 SYSCLKOUT



265   SysCtrlRegs.XCLK.bit.XCLKOUTDIV=2; // Set XCLKOUT = SYSCLKOUT/1



266



267// Peripheral clock enables set for the selected peripherals.



268// If you are not using a peripheral leave the clock off



269// to save on power.



270//



271// Note: not all peripherals are available on all 2802x derivates.



272// Refer to the datasheet for your particular device.



273//



274// This function is not written to be an example of efficient code.



275



276   SysCtrlRegs.PCLKCR0.bit.ADCENCLK = 1;       // ADC



277   SysCtrlRegs.PCLKCR3.bit.COMP1ENCLK = 1;     // COMP1



278   SysCtrlRegs.PCLKCR3.bit.COMP2ENCLK = 1;     // COMP2



279   SysCtrlRegs.PCLKCR3.bit.CPUTIMER0ENCLK = 1; // CPU Timer-0



280   SysCtrlRegs.PCLKCR3.bit.CPUTIMER1ENCLK = 1; // CPU Timer-1



281   SysCtrlRegs.PCLKCR3.bit.CPUTIMER2ENCLK = 1; // CPU Timer-2



282   SysCtrlRegs.PCLKCR1.bit.ECAP1ENCLK = 1;     // eCAP1



283   SysCtrlRegs.PCLKCR1.bit.EPWM1ENCLK = 1;     // EPWM1



284   SysCtrlRegs.PCLKCR1.bit.EPWM2ENCLK = 1;     // EPWM2



285   SysCtrlRegs.PCLKCR1.bit.EPWM3ENCLK = 1;     // EPWM3



286   SysCtrlRegs.PCLKCR1.bit.EPWM4ENCLK = 1;     // EPWM4



287   SysCtrlRegs.PCLKCR3.bit.GPIOINENCLK = 1;    // GPIO



288   SysCtrlRegs.PCLKCR0.bit.HRPWMENCLK=1;       // HRPWM



289   SysCtrlRegs.PCLKCR0.bit.I2CAENCLK = 1;      // I2C



290   SysCtrlRegs.PCLKCR0.bit.SCIAENCLK = 1;      // SCI-A



291   SysCtrlRegs.PCLKCR0.bit.SPIAENCLK = 1;      // SPI-A



292



293   SysCtrlRegs.PCLKCR0.bit.TBCLKSYNC = 1;      // Enable TBCLK within the EPWM



294



295   EDIS;



296}

298//---------------------------------------------------------------------------



299// Example: CsmUnlock:



300//---------------------------------------------------------------------------



301// This function unlocks the CSM. User must replace 0xFFFF's with current



302// password for the DSP. Returns 1 if unlock is successful.



303



304#define STATUS_FAIL          0



305#define STATUS_SUCCESS       1



306



307Uint16 CsmUnlock()



308{



309    volatile Uint16 temp;



310



311    // Load the key registers with the current password. The 0xFFFF's are dummy



312    // passwords.  User should replace them with the correct password for the DSP.



313



314    EALLOW;



315    CsmRegs.KEY0 = 0xFFFF;



316    CsmRegs.KEY1 = 0xFFFF;



317    CsmRegs.KEY2 = 0xFFFF;



318    CsmRegs.KEY3 = 0xFFFF;



319    CsmRegs.KEY4 = 0xFFFF;



320    CsmRegs.KEY5 = 0xFFFF;



321    CsmRegs.KEY6 = 0xFFFF;



322    CsmRegs.KEY7 = 0xFFFF;



323    EDIS;



324



325    // Perform a dummy read of the password locations



326    // if they match the key values, the CSM will unlock



327



328    temp = CsmPwl.PSWD0;



329    temp = CsmPwl.PSWD1;



330    temp = CsmPwl.PSWD2;



331    temp = CsmPwl.PSWD3;



332    temp = CsmPwl.PSWD4;



333    temp = CsmPwl.PSWD5;



334    temp = CsmPwl.PSWD6;



335    temp = CsmPwl.PSWD7;



336



337    // If the CSM unlocked, return succes, otherwise return



338    // failure.



339    if (CsmRegs.CSMSCR.bit.SECURE == 0) return STATUS_SUCCESS;



340    else return STATUS_FAIL;



341



342}

344//---------------------------------------------------------------------------



345// Example: IntOsc1Sel:



346//---------------------------------------------------------------------------



347// This function switches to Internal Oscillator 1 and turns off all other clock



348// sources to minimize power consumption



349



350void IntOsc1Sel (void) {



351    EALLOW;



352    SysCtrlRegs.CLKCTL.bit.INTOSC1OFF = 0;



353    SysCtrlRegs.CLKCTL.bit.OSCCLKSRCSEL=0;  // Clk Src = INTOSC1



354    SysCtrlRegs.CLKCTL.bit.XCLKINOFF=1;     // Turn off XCLKIN



355    SysCtrlRegs.CLKCTL.bit.XTALOSCOFF=1;    // Turn off XTALOSC



356    SysCtrlRegs.CLKCTL.bit.INTOSC2OFF=1;    // Turn off INTOSC2



357    EDIS;



358}

361//---------------------------------------------------------------------------



362// Example: IntOsc2Sel:



363//---------------------------------------------------------------------------



364// This function switches to Internal oscillator 2 from External Oscillator



365// and turns off all other clock sources to minimize power consumption



366// NOTE: If there is no external clock connection, when switching from



367//       INTOSC1 to INTOSC2, EXTOSC and XLCKIN must be turned OFF prior



368//       to switching to internal oscillator 1



369



370void IntOsc2Sel (void) {



371    EALLOW;



372    SysCtrlRegs.CLKCTL.bit.INTOSC2OFF = 0;     // Turn on INTOSC2



373    SysCtrlRegs.CLKCTL.bit.OSCCLKSRC2SEL = 1;  // Switch to INTOSC2



374    SysCtrlRegs.CLKCTL.bit.XCLKINOFF = 1;      // Turn off XCLKIN



375    SysCtrlRegs.CLKCTL.bit.XTALOSCOFF = 1;     // Turn off XTALOSC



376    SysCtrlRegs.CLKCTL.bit.OSCCLKSRCSEL = 1;   // Switch to Internal Oscillator 2/External Oscillator branch



377    SysCtrlRegs.CLKCTL.bit.WDCLKSRCSEL = 0;   //Switch Watchdog Clk Src to internal clock 1



378    SysCtrlRegs.CLKCTL.bit.INTOSC1OFF = 0;     // Turn on INTOSC1



379    EDIS;



380}

382//---------------------------------------------------------------------------



383// Example: XtalOscSel:



384//---------------------------------------------------------------------------



385// This function switches to External CRYSTAL oscillator and turns off all other clock



386// sources to minimize power consumption. This option may not be available on all



387// device packages



388



389void XtalOscSel (void)  {



390     EALLOW;



391     SysCtrlRegs.CLKCTL.bit.XTALOSCOFF = 0;     // Turn on XTALOSC



392     DELAY_US(1000L);                           // 1mS delay to ensure crystal



393                                                // oscillator is up and running.



394                                                // Adjust as needed. 



395     SysCtrlRegs.CLKCTL.bit.XCLKINOFF = 1;      // Turn off XCLKIN



396     SysCtrlRegs.CLKCTL.bit.OSCCLKSRC2SEL = 0;  // Switch to external clock



397     SysCtrlRegs.CLKCTL.bit.OSCCLKSRCSEL = 1;   // Switch from INTOSC1 to INTOSC2/ext clk



398     SysCtrlRegs.CLKCTL.bit.WDCLKSRCSEL = 0;   //Switch Watchdog Clk Src to internal clock 1



399     SysCtrlRegs.CLKCTL.bit.INTOSC2OFF = 1;     // Turn off INTOSC2



400     SysCtrlRegs.CLKCTL.bit.INTOSC1OFF = 0;     // Turn on INTOSC1



401     EDIS;



402



403}

406//---------------------------------------------------------------------------



407// Example: ExtOscSel:



408//---------------------------------------------------------------------------



409// This function switches to External oscillator and turns off all other clock



410// sources to minimize power consumption.



411



412void ExtOscSel (void)  {



413     EALLOW;



414     SysCtrlRegs.XCLK.bit.XCLKINSEL = 1;       // 1-GPIO19 = XCLKIN, 0-GPIO38 = XCLKIN



415     SysCtrlRegs.CLKCTL.bit.XTALOSCOFF = 1;    // Turn on XTALOSC



416     SysCtrlRegs.CLKCTL.bit.XCLKINOFF = 0;     // Turn on XCLKIN



417     SysCtrlRegs.CLKCTL.bit.OSCCLKSRC2SEL = 0; // Switch to external clock



418     SysCtrlRegs.CLKCTL.bit.OSCCLKSRCSEL = 1;  // Switch from INTOSC1 to INTOSC2/ext clk



419     SysCtrlRegs.CLKCTL.bit.WDCLKSRCSEL = 0;   // Switch Watchdog Clk Src to internal clock 1



420     SysCtrlRegs.CLKCTL.bit.INTOSC2OFF = 1;    // Turn off INTOSC2



421     SysCtrlRegs.CLKCTL.bit.INTOSC1OFF = 0;    // Turn on INTOSC1



422     EDIS;



423}
